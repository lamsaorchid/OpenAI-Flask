def get_instagram_account():
    global INSTAGRAM_ACCOUNT_ID
    try:
        url = f"https://graph.facebook.com/v21.0/{PAGE_ID}?fields=instagram_business_account&access_token={PAGE_ACCESS_TOKEN}"
        data = requests.get(url).json()
        if 'instagram_business_account' in data:
            INSTAGRAM_ACCOUNT_ID = data['instagram_business_account']['id']
            logger.info(f"✅ Instagram ID: {INSTAGRAM_ACCOUNT_ID}")
            return True
        return False
    except Exception as e:
        logger.error(f"❌ {e}")
        return False

def bot_worker():
    logger.info("🚀 البوت بدأ...")
    if not get_instagram_account():
        return
    
    while True:
        try:
            url = f"https://graph.facebook.com/v21.0/{INSTAGRAM_ACCOUNT_ID}/media?fields=id&limit=5&access_token={PAGE_ACCESS_TOKEN}"
            posts = requests.get(url).json().get('data', [])
            logger.info(f"✅ {len(posts)} منشورات")
            
            for post in posts:
                c_url = f"https://graph.facebook.com/v21.0/{post['id']}/comments?fields=id,text,username&access_token={PAGE_ACCESS_TOKEN}"
                comments = requests.get(c_url).json().get('data', [])
                
                for c in comments:
                    if c['id'] not in replied:
                        logger.info(f"\n📩 {c['text'][:50]}")
                        reply = get_reply(c['text'])
                        logger.info(f"🤖 {reply[:50]}")
                        
                        r = requests.post(
                            f"https://graph.facebook.com/v21.0/{c['id']}/replies",
                            data={'message': reply, 'access_token': PAGE_ACCESS_TOKEN}
                        )
                        
                        if r.status_code == 200:
                            replied.add(c['id'])
                            stats['total'] += 1
                            logger.info("✅ تم الرد!")
                        else:
                            logger.error(f"❌ {r.text}")
            
            stats['status'] = f'✅ نشط'
            time.sleep(60)
        except Exception as e:
            logger.error(f"❌ {e}")
            time.sleep(120)
